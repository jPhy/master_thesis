\section{Adaptive importance sampling with the variational-Bayes approach}

\subsection{Adaptive importance sampling}

\slide{

    \frametitle{\insertsubsectionhead}

    $$ \int P(x) dx = \int \frac{P(x)}{p(x)} p(x) dx
    \approx \frac{1}{N} \sum^{N}_{n = 1} \frac{P(x_n)}{p(x_n)} \equiv \mu^{{N}} ~~ where ~~ x_n \sim p $$


    \only<2->{squared uncertainty (variance):

    $$ \varmuN = \frac{1}{N} \left[ \int \frac{P(x)}{p(x)} P(x) dx - \left(\int P(x) dx \right)^2 \right] $$}

    \only<3->{\begin{itemize} \item minimize the uncertainty $\varmuN$ with respect to $p$}

        \only<4->{\begin{itemize}
                      \item[--] it suffices to minimize: ~~ $\mathrm{log} \left( \int \frac{P(x)}{p(x)} P(x) dx \right)$}
        \only<5->{    \item[--] by Jensen's inequality:     $\geq \int \mathrm{log} \left( \frac{P(x)}{p(x)} \right) P(x) dx
                                                            \only<6->{\equiv \KLPp}$}
        \only<6->{    \vspace{1.5mm}
                      \item[--] known as the \emph{Kullback-Leibler divergence}}
        \only<4->{\end{itemize}}

    \only<3->{\end{itemize}}
}

\slide{

    \frametitle{\insertsubsectionhead}

    Note:

        \begin{center}
            \begin{tabular}{lcrcc}

                \only<2->{$0 \leq \KLPp$   & and & $  \KLPp = 0$ & $\Leftrightarrow$ & $P=p$ \\}
                \\ % empty row
                \only<3->{$0 \leq \varmuN$ & and & $\varmuN = 0$ & $\Leftrightarrow$ & $P=p$ \\}

            \end{tabular}
        \end{center} \vspace{5mm}

    \only<4->{

      Remember:

        \begin{center}

                \only<4 >{~$\varmuN - const =                         \mathrm{log} \left( \int \frac{P(x)}{p(x)} P(x) dx \right)  \geq \KLPp$}
                \only<5->{ $\varmuN - const = \underbrace{\textstyle{}\mathrm{log} \left( \int \frac{P(x)}{p(x)} P(x) dx \right)} \geq \KLPp$}

        \end{center}

      \only<5->{\vspace{-4mm} \hspace{35mm} \small minimization typically infeasible}

    }

    \only<6->{\vspace{4mm}{}\begin{center}\Large{}\textcolor{red}{minimize $\KLPp$ and hope to approach the unique global minimum $P=p$}{}\end{center}}

}

\subsection{Variational Bayes {\tiny (Gaussian mixture model)}}

\slide{

    \frametitle{\insertsubsectionhead}

    \only<2->{

        restrict $p$ to Gaussian mixtures

        $$ p(x_n | \boldsymbol{\theta}) = \sum_{k=1}^K \pi_k \gauss(x_n | \mu_k, \Sigma_k) , ~~ \boldsymbol{\theta} = \lbrace \pi_k , \mu_k, \Sigma_k \rbrace $$

    }

    \vspace{3mm}

    \only<3->{
        rewrite with \emph{latent variables} $\boldsymbol{Z}$:
        \[z_{nk} =
            \begin{cases}
                \text{1 if }x_n\text{ from component k} \\
                \text{0 else} \\
            \end{cases}
        \]
    }

    \vspace{-13mm}

    \begin{columns}
    \column{0.45\textwidth}

        \only<4->{$$ p(x_n | \boldsymbol{Z} , \boldsymbol{\theta}) = \prod ^K _{k=1} \gauss(x_n | \mu_k, \Sigma_k)^{z_{nk}} $$}
        \only<5->{$$ p(z_{nk}=1 |  \boldsymbol{\theta}) = \pi_k = \prod ^K _{k^\prime=1} \pi_{k^\prime}^{z_{nk^\prime}} $$}

    \column{0.45\textwidth}

        \only<6->{$$ p(x_n | \boldsymbol{\theta}) = \sum_{\boldsymbol{Z}} p(x_n | \boldsymbol{Z} , \boldsymbol{\theta}) p(\boldsymbol{Z} | \boldsymbol{\theta}) $$}

     \end{columns}


}

\slide{

    \frametitle{\insertsubsectionhead}

    \only<2->{without proof} \only<1>{\vspace{4.8mm}}

        \[ \text{minimize} ~ \KLPp \uncover<2->{\Leftrightarrow \text{maximize} ~ \log p(\boldsymbol{X} | \boldsymbol{\theta}) \equiv \log \prod^{N}_{n = 1} p(x_n | \boldsymbol{\theta})} \]

    \only<3->{

        \vspace{3mm}

        impose a \emph{prior} $p(\boldsymbol{\theta})$ and

        rewrite with arbitrary probability distribution $q(\boldsymbol{Z} , \boldsymbol{\theta})$

        \only<4->{\footnotesize \[ \log p(\boldsymbol{X}) = \underbrace{\int \rmdx{\boldsymbol{Z}} \rmdx{\boldsymbol{\theta}} q(\boldsymbol{Z} , \boldsymbol{\theta}) \log \left[\frac{p(\samples, \boldsymbol{Z} , \boldsymbol{\theta})}{q(\boldsymbol{Z} , \boldsymbol{\theta})}\right]} + \underbrace{\int \rmdx{\boldsymbol{Z}} \rmdx{\boldsymbol{\theta}} q(\boldsymbol{Z} , \boldsymbol{\theta}) \log \left[\frac{q(\boldsymbol{Z} , \boldsymbol{\theta})}{p(\boldsymbol{Z} , \boldsymbol{\theta} | \samples)} \right] } \]

            \vspace{-2.5mm}
            ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ $\equiv \mathcal{L}\left[q\right]$ ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ $\equiv \KLqp$

        }

    }

    \only<5->{\Large \textcolor{red}{\vspace{-8mm} \begin{center} assumption $q(\boldsymbol{Z} , \boldsymbol{\theta}) = q(\boldsymbol{Z})q(\boldsymbol{\theta})$ \\ $\Rightarrow$ EM-like algorithm \end{center}}}

}
